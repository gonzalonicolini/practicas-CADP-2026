program parcial;
const
	dimF= 2020;
Type
	compras= record
		cod: integer;
		anio: 1980..2024;
		monto: real;
		dni: integer;
	end;
	lista= ^nodo;
	nodo= record
		dato: compras;
		sig: lista;
	end;
vectorCont= Array [2010..dimF]of real;

procedure cargarLista(var L: lista);
// se dispone

procedure insertarOrdenado(var L: lista; c: compras);
var
	nue, ant, act: lista;
begin
	new(nue);
	nue^.dato:= c;
	if(L = nil)then begin
		nue^.sig:= L;
		L:= nue;
	end else begin
		ant:= L;
		act:= L;
		while(act <> nil)AND(act^.dato.dni < c.dni)do begin
			ant:= act;
			act:= act^.sig;
		end;
		if(ant = act)then begin
			nue^.sig:= L;
			L:= nue;
		end
		else begin
			if(act= nil)then begin
				nue^.sig:= act;
				ant^.sig:= nue;
			end
			else begin
				nue^.sig:= act;
				ant^.sig:= nue;
			end;
		end;
	end;
end;

procedure inicializar(var v: vectorCont);
var
	i: integer;
begin
	for i:= 2010 to dimF do
		v[i]:= 0;
end;

procedure recorrerListaDispo(L: lista; var listaNueva: lista);	// inciso A
begin
	while(L <> nil)do begin
		if(L^.dato.anio >= 2010)AND(L^.dato.anio <= 2020)then
			insertarOrdenado(listaNueva, L^.dato);
		L:= L^.sig;
	end;
end;

procedure verificarMax(unDNI: integer; cantCompras: integer; var max1, max2: integer; var dniMax1, dniMax2: integer);
begin
	if(cantCompras > max1)then begin
		max2:= max1;
		dniMax2:= dniMax1;
		max1:= cantCompras;
		dniMax:= unDNI;
	end else
		if(cantCompras > max2)then begin
			max2:= cantCompras;
			dniMax2:= unDNI;
		end;
end;

procedure buscarMin(ve: vectorCont);
var
	i, posMin: integer;
	min: real;
begin
	min:= 9999;
	for i:= 2010 to dimF do begin
		if(ve[i] < min)then begin
			min:= ve[i];
			posMin:= i;
		end;
	end;
	writeln('El anio con menos facturacion fue ',posMin);
end;

procedure recorrerListaOrdenada(L: lista; var ve: vectorCont);
var
	max1, max2, dniMax1, dniMax2, dniAux, cantCompras: integer;
	montoTotal: real;
begin
	max1:= -1;
	max2:= -1;
	montoTotal:= 0;
	while(L <> nil)do begin
		dniAux:= L^.dato.dni;
		cantCompras:= 0;
		while(L <> nil)AND(L^.dato.dni = dniAux)do begin
			cantCompras:= cantCompras +1;
			ve[L^.dato.anio]:= ve[L^.dato.anio] + L^.dato.monto;
			if(L^.dato.cod MOD 10 = 0)then
				montoTotal:= montoTotal + L^.dato.monto;
			L:= L^.sig;
		end;
		verificarMax(dniAux, cantCompras, max1, max2, dniMax1, dniMax2);
	end;
	writeln('El Dni con mas compras es ',dniMax1);
	writeln('El segundo dni con mas compras es ',dniMax2);
	writeln('El monto total de compras con codigo multiplo de 10 es ',montoTotal);
	buscarMin(ve);
end;

var
	pri, listaDos: lista;
	vCont: vectorCont;
begin
	pri:= nil;
	listaDos:= nil;
	inicializar(vCont);
	cargarLista(pri); 		// se dispone
	recorrerListaDispo(pri, listaDos);
	recorrerListaOrdenada(listaDos, vCont);	// hace punto B
end.
