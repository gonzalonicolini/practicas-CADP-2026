function cumple(num:integer):boolean;
var
	digito:integer;
	ok:boolean;
	par:integer;
begin
	par:= 0;
	ok:= false;
	while (num <> 0) AND (par <= 4) do begin
		dig:= num MOD 10;
		if(dig MOD 2 = 0) then
			par:= par + 1;
		num:= num DIV 10;
	end;
	if(par = 4) then
		ok:= true;
	cumple:= ok;
end;


procedure insertarOrdenado(var l2:lista; li:lbros);
var
	ant,act,nue:lista;
begin
	new(nue);
	nue^.dato:= li;
	ant:= l2;
	act:= l2;
	while(act <> nil) and (li.autor > act^.dato.autor) do begin
		ant:= act;
		act:= act^.sig;
	end;
	if(act = l)then
		l2:= nue
	else
		ant^.sig:= nue;
	nue^.sig:= act;
end;
