program parcial;
const dimF= 2024;
Type
	subr: 1980..2024;
	compras= record
		cod: integer;
		anio: subr;
		monto: real;
		dni: integer;
	end;
	lista= ^nodo;
	nodo= record
		dato: compras;
		sig: lista;
	end;
	vecCont= Array[subr]of real;
	
procedure armarLista(var L: lista);	// se dispone

procedure insertarOrdenado(var L: lista; c: compras);
var
	nue, ant, act: lista;
begin
	new(nue);
	nue^.dato: c;
	act:= L;
	ant:= L;
	while(act <> nil)and(act^.dato.dni < c.dni)do begin
		ant:= act;
		act:= act^.sig;
	end;
	if(ant = act)then begin
		nue^.sig:= L;
		L:= nue;
	end
	else
		nue^.sig:= act;
	ant^.sig:= nue;
end;

procedure inicializarCont(var ve: vecCont);
var
	i: integer;
begin
	for i:= 1980 to dimF do
		ve[i]:= 0;
end;

procedure recorrerLista(L: lista; var listaNueva: lista);
begin
	while(L <> nil)do begin
		if(L^.dato.anio >= 1980)AND(L^.dato.anio <=2024)then
			insertarOrdenado(listaNueva, L^.dato);
		L:= L^.sig;
	end;
end;

procedure verificarMaximos(unDni: integer; unaCantCompras: integer; var max1, max2, dniMax1, dniMax2: integer);
begin
	if(unaCantCompras > max1)then begin
		max2:= max1;
		dniMax2:= dniMax1;
		max1:= unaCantCompras;
		dniMax1:= unDni;
	end else
		if(unaCantCompras > max2)then begin
			max2:= unaCantCompras;
			dniMax2:= unDni;
		end;
end;

procedure recorrerNueva(L: lista; var ve: vecCont);
var
	dniAux, cantAux, max1, max2, dniMax2, dniMax1: integer;
	montoTotal: real;
begin
	max1:= -1;
	max2:= -1;
	inicializarCont(ve);
	montoTotal:= 0;
	while(L <> nil)do begin
		dniAux:= L^.dato.dni;
		cantAux:= 0;
		while(L <> nil)AND(dniAux = L^.dato.dni)do begin
			cantAux:= cantAux +1;
			ve[L^.dato.anio]:= ve[L^.dato.anio] + L^.dato.monto;
			if(L^.dato.cod MOD 10 = 0)then
				montoTotal:= montoTotal + L^.dato.monto;
			L:= L^.sig;
		end;
		verificarMaximos(dniAux, cantAux, max1, max2, dniMax1, dniMax2);
	end;
	writeln('El dni que mas compro es el ',dniMax1);	// inciso B 2
	writeln('El segundo que mas compro es el ',dniMax2);
	writeln('El monto total de compras con codigo multiplo de 10 es de ',montoTotal);
	// inciso B 3
end;

procedure buscarMin(ve: vecCont);
var
	i, posMin: integer;
	min: real;
begin
	min:= 9999;
	for i:= 1980 to dimF do
		if(ve[i] < min)then begin
			min:= ve[i];
			posMin:= i;
		end;
	writeln('El anio con menor facturacion es ',posMin);	// 2A
end;

var
	pri, listaNueva: lista;
	cont: vecCont;
Begin
	pri:= nil;
	listaNueva:= nil;
	armarLista(pri);	// se dispone
	recorrerLista(pri, listaNueva);	// inciso A
	recorrerNueva(listaNueva, cont);	// incicos 2B y 2C
	buscarMin(cont);	// inciso 2A
end.
